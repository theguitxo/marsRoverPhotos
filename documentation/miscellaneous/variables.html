Default value : createSelector ( selectState, selectRouteParams, (state: StoreState, params: Params): boolean => { const rover = state?.roversList?.find(roverItem => roverItem.code === params[ROVER_CONSTANTS.DETAILS_PARAMS.CODE]); const camsRover = rover?.photos?.find(photosRover => photosRover.sol === +params[ROVER_CONSTANTS.DETAILS_PARAMS.SOL])?.cameras; const camsRoverPhotos = Array.from(new Set(rover?.roverPhotos?. filter(roverSol => roverSol.sol === +params[ROVER_CONSTANTS.DETAILS_PARAMS.SOL]). map(roverPhoto => roverPhoto.camera))); if (!(!!params[ROVER_CONSTANTS.DETAILS_PARAMS.CAMERA])) { return (camsRover?.length === camsRoverPhotos?.length); } return (camsRoverPhotos?.indexOf(params[ROVER_CONSTANTS.DETAILS_PARAMS.CAMERA]) > -1); } )
